$ - интерполяция (замена плюса)
    пример

    const category = "toy";
    console.log(`https://someurl/${category}/5`); важное условие. кавычки в таком случае выставляются тильдой (буква ё на клаве)



++ - инкримент ( увеличивает на 1)
    пример 

    let incr = 10,
    incr++; (постфиксная форма. записана после.)
    ++incr; (префиксаня форма. записана перед.)
    разница

    let x = 3; 
    let y = ++x; - выдаст у = 4, х = 4
    let x = 3; 
    let y = x++; - выдаст у = 3, х = 4



-- - дикримент ( уменшает на 1)

    пример 
    let decr = 10,
    decr--; (постфиксная форма. записана после.)
    --decr; (префиксаня форма. записана перед.)
    разница

    console.log(decr--); - не работает сама по себе. только с коментарием выше.
    console.log(--decr); - работает и так

    % - остаток от деления двух чисел.
    пример

    console.log(20%3); - выведет остаток от деления - 2



&& - оператор "и"
|| - оператор "или"
! - оператор "не". обращает в противоположное (
    let toys = true;
    console.log(!toys) - вернёт false)



== - сравнивает значения. но не сравнивает по типу данных
    8 == "8"; - true

=== - сравнивает и значение и тип данных
    8 === "8"; - false
    интересно!
    console.log(2 + 2 * 2 != '6') - false
    console.log(2 + 2 * 2 !== '6') - true - оператор заменяет 3-е "="



if - условие. 
    пример

        let num = 4;
        if (num == 9) {
            console.log('Ok!');
        } else {
            console.log('Error!');
        }
        (если условий больше, то else if и дальше как будто if)

    точно то же (тернарный оператор): 
        (num === 4) ? console.log('Ok!') {что делать если верно} : console.log('Error') {что делать если не верно};

    точно то же (если много ветвлений. работает на строгое сравнение. при num = '50' выдаст значение дефолт): 
        const num = 50;

        switch (num) {
            case 49:
                console.log('неверно');
                break;
            case 100:
                console.log('неверно');
                break;
            case 51:
                console.log('правильно');
                break;
            default:
                console.log('не в этот раз');
                break;
        }



while - делать до тех пор пока не выполниться
    пример
    let num = 50;

    while (num < 55) {
        console.log(num);
        num++
    }

    тоже самое:

    let num = 50;

    do {
        сonsole.log(num);
        num++;
    }

    while (num <55);



for - делать указанное количество раз или до тех пор пока не выполниться действие
    пример

    (делать до тех пор пока і не будет = 8)
    for (let i = 1; i < 8; i++) {
        console.log(num);
        num++;
    }



break - прерывает что то, например цикл
    пример с бесконечным циклом
    for (let i = 1; i < 0; i++) {
        if (i === 10){
            break;
        }
        console.log(i);
        i++;
    } - остановиться после 10 повторений



continue - пропускает заданный шпаргалка
    пример 
    for (let i = 1; i < 10; i++) {
        if (i === 6){
            continue;
        }
        console.log(i);
        i++;
    } - пропустит 6 шаг и пойдёт дальше. ... 4, 5, 7, 8....
